    <script>
        // --- WEBRTC CONFIGURATION ---
        // Basic configuration needed for peer connection
        const pcConfig = {
            iceServers: [{ urls: "stun:stun.l.google.com:19302" }] 
        };
        
        let pc = null;
        let localStream = null;
        let audioContext = null;
        let startTime = 0;

        // UI Elements
        const logBox = document.getElementById('sys-log');
        const statusText = document.getElementById('status-text');
        const reactor = document.getElementById('reactor');
        const connectionStatus = document.getElementById('connection-status');
        const pingDot = document.getElementById('ping-dot');

        async function startJarvis() {
            document.getElementById('init-overlay').classList.add('opacity-0', 'pointer-events-none');
            document.getElementById('hud').classList.remove('opacity-0', 'blur-sm');
            updateClock(); setInterval(updateClock, 1000);
            
            log("Initializing WebRTC Protocol...");
            await startWebRTC();
        }

        async function startWebRTC() {
            try {
                // 1. Get User Media (Microphone)
                localStream = await navigator.mediaDevices.getUserMedia({
                    audio: {
                        echoCancellation: true,
                        noiseSuppression: true,
                        autoGainControl: true,
                        channelCount: 1,
                        sampleRate: 48000 // Browser usually handles this, but requesting 48k is safe
                    },
                    video: false
                });
                
                log("Microphone Access Granted.");
                setupVisualizer(localStream); // Keep the cool visualizer

                // 2. Create Peer Connection
                pc = new RTCPeerConnection(pcConfig);

                // 3. Add Local Track to PC
                localStream.getTracks().forEach(track => {
                    pc.addTrack(track, localStream);
                });

                // 4. Handle Incoming Remote Track (JARVIS Voice)
                pc.ontrack = (event) => {
                    log("Receiving Audio Stream from JARVIS...");
                    const remoteAudio = document.getElementById('remote-audio');
                    if (remoteAudio.srcObject !== event.streams[0]) {
                        remoteAudio.srcObject = event.streams[0];
                        log("ðŸ”Š Audio Output Connected.");
                        statusText.innerText = "ONLINE";
                        reactor.classList.add('status-connected');
                        pingDot.classList.remove('bg-red-500');
                        pingDot.classList.add('bg-green-500', 'animate-ping');
                        connectionStatus.innerText = "SECURE LINK ESTABLISHED";
                    }
                };

                // 5. Create Offer & Negotiate
                const offer = await pc.createOffer();
                await pc.setLocalDescription(offer);

                log("Sending SDP Offer to Server...");
                startTime = Date.now();

                // Wait for ICE gathering to complete (Simple approach for local dev)
                await new Promise((resolve) => {
                    if (pc.iceGatheringState === 'complete') {
                        resolve();
                    } else {
                        pc.onicecandidate = (event) => {
                            if (event.candidate === null) {
                                resolve();
                            }
                        };
                    }
                });

                // 6. Send Offer to Python Backend
                const response = await fetch("/offer", {
                    body: JSON.stringify({
                        sdp: pc.localDescription.sdp,
                        type: pc.localDescription.type,
                    }),
                    headers: { "Content-Type": "application/json" },
                    method: "POST",
                });

                const answer = await response.json();
                log("Received SDP Answer from Server.");
                
                // 7. Set Remote Description
                await pc.setRemoteDescription(new RTCSessionDescription(answer));
                
                const latency = Date.now() - startTime;
                document.getElementById('latency-val').innerText = latency + " ms";
                log(`Handshake Complete. Latency: ${latency}ms`);

            } catch (err) {
                log("âŒ CONNECTION ERROR: " + err);
                statusText.innerText = "SYSTEM FAILURE";
                reactor.classList.add('status-error');
            }
        }

        // --- UTILITIES ---
        function log(msg) {
            const p = document.createElement('p');
            p.innerText = `> ${msg}`;
            logBox.appendChild(p);
            logBox.scrollTop = logBox.scrollHeight;
        }

        function updateClock() {
            document.getElementById('clock').innerText = new Date().toLocaleTimeString();
        }

        function setupVisualizer(stream) {
            if (!audioContext) audioContext = new (window.AudioContext || window.webkitAudioContext)();
            const canvas = document.getElementById("visualizer");
            const ctx = canvas.getContext("2d");
            const source = audioContext.createMediaStreamSource(stream);
            const analyser = audioContext.createAnalyser();
            
            source.connect(analyser);
            analyser.fftSize = 256;
            
            const bufferLength = analyser.frequencyBinCount;
            const dataArray = new Uint8Array(bufferLength);

            function draw() {
                requestAnimationFrame(draw);
                analyser.getByteFrequencyData(dataArray);
                ctx.clearRect(0, 0, canvas.width, canvas.height);
                
                const barWidth = (canvas.width / bufferLength) * 2.5;
                let x = 0;
                
                for(let i = 0; i < bufferLength; i++) {
                    const barHeight = dataArray[i] / 2;
                    ctx.fillStyle = `rgb(0, 255, 255)`; // Cyan color
                    ctx.fillRect(x, canvas.height - barHeight, barWidth, barHeight);
                    x += barWidth + 1;
                }
            }
            draw();
        }
